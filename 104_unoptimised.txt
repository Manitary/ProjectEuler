k:=2749;
check:=false;
repeat
	k+:=1;
	n:=Fibonacci(k);
	s:=IntegerToString(n);
	h:={StringToInteger(i): i in SequenceToSet(Eltseq(Substring(s,1,9)))};
	t:={StringToInteger(i): i in SequenceToSet(Eltseq(Substring(s,#s-8,9)))};
	if h eq {1..9} and t eq {1..9} then
		k;
		check:=true;
	end if;
until check;
